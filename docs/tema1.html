import numpy as np
import matplotlib.pyplot as plt

P = np.array([1, 2, 3, 4, 5, 6, 7])

R = 8.31 # Gasskonstanten
T = 298  # Temp. i K
n = 1    # Stoffmengde i mol

V = n*R*T/P

plt.plot(P,V,color='hotpink',marker='o',linestyle='')
plt.show()
